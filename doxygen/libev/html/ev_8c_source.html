01722 /* find a suitable new size for the given array, */
01723 /* hopefully by rounding to a nice-to-malloc size */
01724 inline_size int
01725 array_nextsize (int elem, int cur, int cnt)
01726 {
01727   int ncur = cur + 1;
01728 
01729   do
01730     ncur <<= 1;
01731   while (cnt > ncur);

01733   /* if size is large, round to MALLOC_ROUND - 4 * longs to accommodate malloc overhead */
01734   if (elem * ncur > MALLOC_ROUND - sizeof (void *) * 4)
01735     {
01736       ncur *= elem;
01737       ncur = (ncur + elem + (MALLOC_ROUND - 1) + sizeof (void *) * 4) & ~(MALLOC_ROUND - 1);
01738       ncur = ncur - sizeof (void *) * 4;
01739       ncur /= elem;
01740     }

02020 /*
02021  * the heap functions want a real array index. array index 0 is guaranteed to not
02022  * be in-use at any time. the first heap entry is at array [HEAP0]. DHEAP gives
02023  * the branching factor of the d-tree.
02024  */
02025 
02026 /*
02027  * at the moment we allow libev the luxury of two heaps,
02028  * a small-code-size 2-heap one and a ~1.5kb larger 4-heap
02029  * which is more cache-efficient.
02030  * the difference is about 5% with 50000+ watchers.
02031  */
02032 #if EV_USE_4HEAP
02033 
02034 #define DHEAP 4
02035 #define HEAP0 (DHEAP - 1) /* index of first element in heap */
02036 #define HPARENT(k) ((((k) - HEAP0 - 1) / DHEAP) + HEAP0)
02037 #define UPHEAP_DONE(p,k) ((p) == (k))

02039 /* away from the root */
02040 inline_speed void
02041 downheap (ANHE *heap, int N, int k)
02042 {
02043   ANHE he = heap [k];
02044   ANHE *E = heap + N + HEAP0;
02045 
02046   for (;;)
02047     {
02048       ev_tstamp minat;
02049       ANHE *minpos;
02050       ANHE *pos = heap + DHEAP * (k - HEAP0) + HEAP0 + 1;
02051 
02052       /* find minimum child */
02053       if (expect_true (pos + DHEAP - 1 < E))
02054         {
02055           /* fast path */                               (minpos = pos + 0), (minat = ANHE_at (*minpos));
02056           if (               ANHE_at (pos [1]) < minat) (minpos = pos + 1), (minat = ANHE_at (*minpos));
02057           if (               ANHE_at (pos [2]) < minat) (minpos = pos + 2), (minat = ANHE_at (*minpos));
02058           if (               ANHE_at (pos [3]) < minat) (minpos = pos + 3), (minat = ANHE_at (*minpos));
02059         }
02060       else if (pos < E)
02061         {
02062           /* slow path */                               (minpos = pos + 0), (minat = ANHE_at (*minpos));
02063           if (pos + 1 < E && ANHE_at (pos [1]) < minat) (minpos = pos + 1), (minat = ANHE_at (*minpos));
02064           if (pos + 2 < E && ANHE_at (pos [2]) < minat) (minpos = pos + 2), (minat = ANHE_at (*minpos));
02065           if (pos + 3 < E && ANHE_at (pos [3]) < minat) (minpos = pos + 3), (minat = ANHE_at (*minpos));
02066         }
02067       else
02068         break;
02069 
02070       if (ANHE_at (he) <= minat)
02071         break;
02072 
02073       heap [k] = *minpos;
02074       ev_active (ANHE_w (*minpos)) = k;
02075 
02076       k = minpos - heap;
02077     }
02078 
02079   heap [k] = he;
02080   ev_active (ANHE_w (he)) = k;
02081 }
02082 
02083 #else /* 4HEAP */

02086 #define HPARENT(k) ((k) >> 1)
02087 #define UPHEAP_DONE(p,k) (!(p))

02089 /* away from the root */
02090 inline_speed void
02091 downheap (ANHE *heap, int N, int k)
02092 {
02093   ANHE he = heap [k];
02094 
02095   for (;;)
02096     {
02097       int c = k << 1;
02098 
02099       if (c >= N + HEAP0)
02100         break;
02101 
02102       c += c + 1 < N + HEAP0 && ANHE_at (heap [c]) > ANHE_at (heap [c + 1])
02103            ? 1 : 0;
02104 
02105       if (ANHE_at (he) <= ANHE_at (heap [c]))
02106         break;
02107 
02108       heap [k] = heap [c];
02109       ev_active (ANHE_w (heap [k])) = k;
02110       
02111       k = c;
02112     }
02113 
02114   heap [k] = he;
02115   ev_active (ANHE_w (he)) = k;
02116 }
02117 #endif

02122 {
02123   ANHE he = heap [k];
02124 
02125   for (;;)
02126     {
02127       int p = HPARENT (k);
02128 
02129       if (UPHEAP_DONE (p, k) || ANHE_at (heap [p]) <= ANHE_at (he))
02130         break;
02131 
02132       heap [k] = heap [p];
02133       ev_active (ANHE_w (heap [k])) = k;
02134       k = p;
02135     }
02136 
02137   heap [k] = he;
02138   ev_active (ANHE_w (he)) = k;
02139 }

03263 /* fetch new monotonic and realtime times from the kernel */
03264 /* also detect if there was a timejump, and act accordingly */
03265 inline_speed void
03266 time_update (EV_P_ ev_tstamp max_block)
03267 {
03268 #if EV_USE_MONOTONIC
03269   if (expect_true (have_monotonic))
03270     {
03271       int i;
03272       ev_tstamp odiff = rtmn_diff;
03273 
03274       mn_now = get_clock ();
03275 
03276       /* only fetch the realtime clock every 0.5*MIN_TIMEJUMP seconds */
03277       /* interpolate in the meantime */
03278       if (expect_true (mn_now - now_floor < MIN_TIMEJUMP * .5))
03279         {
03280           ev_rt_now = rtmn_diff + mn_now;
03281           return;
03282         }
03283 
03284       now_floor = mn_now;
03285       ev_rt_now = ev_time ();
03286 
03287       /* loop a few times, before making important decisions.
03288        * on the choice of "4": one iteration isn't enough,
03289        * in case we get preempted during the calls to
03290        * ev_time and get_clock. a second call is almost guaranteed
03291        * to succeed in that case, though. and looping a few more times
03292        * doesn't hurt either as we only do this on time-jumps or
03293        * in the unlikely event of having been preempted here.
03294        */
03295       for (i = 4; --i; )
03296         {
03297           ev_tstamp diff;
03298           rtmn_diff = ev_rt_now - mn_now;
03299 
03300           diff = odiff - rtmn_diff;
03301 
03302           if (expect_true ((diff < 0. ? -diff : diff) < MIN_TIMEJUMP))
03303             return; /* all is well */
03304 
03305           ev_rt_now = ev_time ();
03306           mn_now    = get_clock ();
03307           now_floor = mn_now;
03308         }
03309 
03310       /* no timer adjustment, as the monotonic clock doesn't jump */
03311       /* timers_reschedule (EV_A_ rtmn_diff - odiff) */
03312 # if EV_PERIODIC_ENABLE
03313       periodics_reschedule (EV_A);
03314 # endif
03315     }
03316   else
03317 #endif
03318     {
03319       ev_rt_now = ev_time ();
03320 
03321       if (expect_false (mn_now > ev_rt_now || ev_rt_now > mn_now + max_block + MIN_TIMEJUMP))
03322         {
03323           /* adjust timers. this is easy, as the offset is the same for all of them */
03324           timers_reschedule (EV_A_ ev_rt_now - mn_now);
03325 #if EV_PERIODIC_ENABLE
03326           periodics_reschedule (EV_A);
03327 #endif
03328         }
03329 
03330       mn_now = ev_rt_now;
03331     }
03332 }

03334 int
03335 ev_run (EV_P_ int flags)
03336 {
03337 #if EV_FEATURE_API
03338   ++loop_depth;
03339 #endif
03340 
03341   assert (("libev: ev_loop recursion during release detected", loop_done != EVBREAK_RECURSE));
03342 
03343   loop_done = EVBREAK_CANCEL;
03344 
03345   EV_INVOKE_PENDING; /* in case we recurse, ensure ordering stays nice and clean */
03346 
03347   do
03348     {
03349 #if EV_VERIFY >= 2
03350       ev_verify (EV_A);
03351 #endif
03352 
03353 #ifndef _WIN32
03354       if (expect_false (curpid)) /* penalise the forking check even more */
03355         if (expect_false (getpid () != curpid))
03356           {
03357             curpid = getpid ();
03358             postfork = 1;
03359           }
03360 #endif
03361 
03362 #if EV_FORK_ENABLE
03363       /* we might have forked, so queue fork handlers */
03364       if (expect_false (postfork))
03365         if (forkcnt)
03366           {
03367             queue_events (EV_A_ (W *)forks, forkcnt, EV_FORK);
03368             EV_INVOKE_PENDING;
03369           }
03370 #endif
03371 
03372 #if EV_PREPARE_ENABLE
03373       /* queue prepare watchers (and execute them) */
03374       if (expect_false (preparecnt))
03375         {
03376           queue_events (EV_A_ (W *)prepares, preparecnt, EV_PREPARE);
03377           EV_INVOKE_PENDING;
03378         }
03379 #endif
03380 
03381       if (expect_false (loop_done))
03382         break;
03383 
03384       /* we might have forked, so reify kernel state if necessary */
03385       if (expect_false (postfork))
03386         loop_fork (EV_A);
03387 
03388       /* update fd-related kernel structures */
03389       fd_reify (EV_A);
03390 
03391       /* calculate blocking time */
03392       {
03393         ev_tstamp waittime  = 0.;
03394         ev_tstamp sleeptime = 0.;
03395 
03396         /* remember old timestamp for io_blocktime calculation */
03397         ev_tstamp prev_mn_now = mn_now;
03398 
03399         /* update time to cancel out callback processing overhead */
03400         time_update (EV_A_ 1e100);
03401 
03402         /* from now on, we want a pipe-wake-up */
03403         pipe_write_wanted = 1;
03404 
03405         ECB_MEMORY_FENCE; /* make sure pipe_write_wanted is visible before we check for potential skips */
03406 
03407         if (expect_true (!(flags & EVRUN_NOWAIT || idleall || !activecnt || pipe_write_skipped)))
03408           {
03409             waittime = MAX_BLOCKTIME;
03410 
03411             if (timercnt)
03412               {
03413                 ev_tstamp to = ANHE_at (timers [HEAP0]) - mn_now;
03414                 if (waittime > to) waittime = to;
03415               }
03416 
03417 #if EV_PERIODIC_ENABLE
03418             if (periodiccnt)
03419               {
03420                 ev_tstamp to = ANHE_at (periodics [HEAP0]) - ev_rt_now;
03421                 if (waittime > to) waittime = to;
03422               }
03423 #endif
03424 
03425             /* don't let timeouts decrease the waittime below timeout_blocktime */
03426             if (expect_false (waittime < timeout_blocktime))
03427               waittime = timeout_blocktime;
03428 
03429             /* at this point, we NEED to wait, so we have to ensure */
03430             /* to pass a minimum nonzero value to the backend */
03431             if (expect_false (waittime < backend_mintime))
03432               waittime = backend_mintime;
03433 
03434             /* extra check because io_blocktime is commonly 0 */
03435             if (expect_false (io_blocktime))
03436               {
03437                 sleeptime = io_blocktime - (mn_now - prev_mn_now);
03438 
03439                 if (sleeptime > waittime - backend_mintime)
03440                   sleeptime = waittime - backend_mintime;
03441 
03442                 if (expect_true (sleeptime > 0.))
03443                   {
03444                     ev_sleep (sleeptime);
03445                     waittime -= sleeptime;
03446                   }
03447               }
03448           }
03449 
03450 #if EV_FEATURE_API
03451         ++loop_count;
03452 #endif
03453         assert ((loop_done = EVBREAK_RECURSE, 1)); /* assert for side effect */
03454         backend_poll (EV_A_ waittime);
03455         assert ((loop_done = EVBREAK_CANCEL, 1)); /* assert for side effect */
03456 
03457         pipe_write_wanted = 0; /* just an optimisation, no fence needed */
03458 
03459         ECB_MEMORY_FENCE_ACQUIRE;
03460         if (pipe_write_skipped)
03461           {
03462             assert (("libev: pipe_w not active, but pipe not written", ev_is_active (&pipe_w)));
03463             ev_feed_event (EV_A_ &pipe_w, EV_CUSTOM);
03464           }
03465 
03466 
03467         /* update ev_rt_now, do magic */
03468         time_update (EV_A_ waittime + sleeptime);
03469       }
03470 
03471       /* queue pending timers and reschedule them */
03472       timers_reify (EV_A); /* relative timers called last */
03473 #if EV_PERIODIC_ENABLE
03474       periodics_reify (EV_A); /* absolute timers called first */
03475 #endif
03476 
03477 #if EV_IDLE_ENABLE
03478       /* queue idle watchers unless other events are pending */
03479       idle_reify (EV_A);
03480 #endif
03481 
03482 #if EV_CHECK_ENABLE
03483       /* queue check watchers, to be executed first */
03484       if (expect_false (checkcnt))
03485         queue_events (EV_A_ (W *)checks, checkcnt, EV_CHECK);
03486 #endif
03487 
03488       EV_INVOKE_PENDING;
03489     }
03490   while (expect_true (
03491     activecnt
03492     && !loop_done
03493     && !(flags & (EVRUN_ONCE | EVRUN_NOWAIT))
03494   ));
03495 
03496   if (loop_done == EVBREAK_ONE)
03497     loop_done = EVBREAK_CANCEL;
03498 
03499 #if EV_FEATURE_API
03500   --loop_depth;
03501 #endif
03502 
03503   return activecnt;
03504 }
