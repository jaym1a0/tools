00040 /*
00041  * general notes about epoll:
00042  *
00043  * a) epoll silently removes fds from the fd set. as nothing tells us
00044  *    that an fd has been removed otherwise, we have to continually
00045  *    "rearm" fds that we suspect *might* have changed (same
00046  *    problem with kqueue, but much less costly there).
00047  * b) the fact that ADD != MOD creates a lot of extra syscalls due to a)
00048  *    and seems not to have any advantage.
00049  * c) the inability to handle fork or file descriptors (think dup)
00050  *    limits the applicability over poll, so this is not a generic
00051  *    poll replacement.
00052  * d) epoll doesn't work the same as select with many file descriptors
00053  *    (such as files). while not critical, no other advanced interface
00054  *    seems to share this (rather non-unixy) limitation.
00055  * e) epoll claims to be embeddable, but in practise you never get
00056  *    a ready event for the epoll fd (broken: <=2.6.26, working: >=2.6.32).
00057  * f) epoll_ctl returning EPERM means the fd is always ready.
00058  *
00059  * lots of "weird code" and complication handling in this file is due
00060  * to these design problems with epoll, as we try very hard to avoid
00061  * epoll_ctl syscalls for common usage patterns and handle the breakage
00062  * ensuing from receiving events for closed and otherwise long gone
00063  * file descriptors.
00064  */

00070 static void
00071 epoll_modify (EV_P_ int fd, int oev, int nev)
00072 {
00073   struct epoll_event ev;
00074   unsigned char oldmask;
00075 
00076   /*
00077    * we handle EPOLL_CTL_DEL by ignoring it here
00078    * on the assumption that the fd is gone anyways
00079    * if that is wrong, we have to handle the spurious
00080    * event in epoll_poll.
00081    * if the fd is added again, we try to ADD it, and, if that
00082    * fails, we assume it still has the same eventmask.
00083    */
00084   if (!nev)
00085     return;
00086 
00087   oldmask = anfds [fd].emask;
00088   anfds [fd].emask = nev;
00089 
00090   /* store the generation counter in the upper 32 bits, the fd in the lower 32 bits */
00091   ev.data.u64 = (uint64_t)(uint32_t)fd
00092               | ((uint64_t)(uint32_t)++anfds [fd].egen << 32);
00093   ev.events   = (nev & EV_READ  ? EPOLLIN  : 0)
00094               | (nev & EV_WRITE ? EPOLLOUT : 0);
00095 
00096   if (expect_true (!epoll_ctl (backend_fd, oev && oldmask != nev ? EPOLL_CTL_MOD : EPOLL_CTL_ADD, fd, &ev)))
00097     return;
00098 
00099   if (expect_true (errno == ENOENT))
00100     {
00101       /* if ENOENT then the fd went away, so try to do the right thing */
00102       if (!nev)
00103         goto dec_egen;
00104 
00105       if (!epoll_ctl (backend_fd, EPOLL_CTL_ADD, fd, &ev))
00106         return;
00107     }
00108   else if (expect_true (errno == EEXIST))
00109     {
00110       /* EEXIST means we ignored a previous DEL, but the fd is still active */
00111       /* if the kernel mask is the same as the new mask, we assume it hasn't changed */
00112       if (oldmask == nev)
00113         goto dec_egen;
00114 
00115       if (!epoll_ctl (backend_fd, EPOLL_CTL_MOD, fd, &ev))
00116         return;
00117     }
00118   else if (expect_true (errno == EPERM))
00119     {
00120       /* EPERM means the fd is always ready, but epoll is too snobbish */
00121       /* to handle it, unlike select or poll. */
00122       anfds [fd].emask = EV_EMASK_EPERM;
00123 
00124       /* add fd to epoll_eperms, if not already inside */
00125       if (!(oldmask & EV_EMASK_EPERM))
00126         {
00127           array_needsize (int, epoll_eperms, epoll_epermmax, epoll_epermcnt + 1, EMPTY2);
00128           epoll_eperms [epoll_epermcnt++] = fd;
00129         }
00130 
00131       return;
00132     }
00133 
00134   fd_kill (EV_A_ fd);
00135 
00136 dec_egen:
00137   /* we didn't successfully call epoll_ctl, so decrement the generation counter again */
00138   --anfds [fd].egen;
00139 }

00141 static void
00142 epoll_poll (EV_P_ ev_tstamp timeout)
00143 {
00144   int i;
00145   int eventcnt;
00146 
00147   if (expect_false (epoll_epermcnt))
00148     timeout = 0.;
00149 
00150   /* epoll wait times cannot be larger than (LONG_MAX - 999UL) / HZ msecs, which is below */
00151   /* the default libev max wait time, however. */
00152   EV_RELEASE_CB;
00153   eventcnt = epoll_wait (backend_fd, epoll_events, epoll_eventmax, timeout * 1e3);
00154   EV_ACQUIRE_CB;
00155 
00156   if (expect_false (eventcnt < 0))
00157     {
00158       if (errno != EINTR)
00159         ev_syserr ("(libev) epoll_wait");
00160 
00161       return;
00162     }
00163 
00164   for (i = 0; i < eventcnt; ++i)
00165     {
00166       struct epoll_event *ev = epoll_events + i;
00167 
00168       int fd = (uint32_t)ev->data.u64; /* mask out the lower 32 bits */
00169       int want = anfds [fd].events;
00170       int got  = (ev->events & (EPOLLOUT | EPOLLERR | EPOLLHUP) ? EV_WRITE : 0)
00171                | (ev->events & (EPOLLIN  | EPOLLERR | EPOLLHUP) ? EV_READ  : 0);
00172 
00173       /*
00174        * check for spurious notification.
00175        * this only finds spurious notifications on egen updates
00176        * other spurious notifications will be found by epoll_ctl, below
00177        * we assume that fd is always in range, as we never shrink the anfds array
00178        */
00179       if (expect_false ((uint32_t)anfds [fd].egen != (uint32_t)(ev->data.u64 >> 32)))
00180         {
00181           /* recreate kernel state */
00182           postfork = 1;
00183           continue;
00184         }
00185 
00186       if (expect_false (got & ~want))
00187         {
00188           anfds [fd].emask = want;
00189 
00190           /*
00191            * we received an event but are not interested in it, try mod or del
00192            * this often happens because we optimistically do not unregister fds
00193            * when we are no longer interested in them, but also when we get spurious
00194            * notifications for fds from another process. this is partially handled
00195            * above with the gencounter check (== our fd is not the event fd), and
00196            * partially here, when epoll_ctl returns an error (== a child has the fd
00197            * but we closed it).
00198            */
00199           ev->events = (want & EV_READ  ? EPOLLIN  : 0)
00200                      | (want & EV_WRITE ? EPOLLOUT : 0);
00201 
00202           /* pre-2.6.9 kernels require a non-null pointer with EPOLL_CTL_DEL, */
00203           /* which is fortunately easy to do for us. */
00204           if (epoll_ctl (backend_fd, want ? EPOLL_CTL_MOD : EPOLL_CTL_DEL, fd, ev))
00205             {
00206               postfork = 1; /* an error occurred, recreate kernel state */
00207               continue;
00208             }
00209         }
00210 
00211       fd_event (EV_A_ fd, got);
00212     }
00213 
00214   /* if the receive array was full, increase its size */
00215   if (expect_false (eventcnt == epoll_eventmax))
00216     {
00217       ev_free (epoll_events);
00218       epoll_eventmax = array_nextsize (sizeof (struct epoll_event), epoll_eventmax, epoll_eventmax + 1);
00219       epoll_events = (struct epoll_event *)ev_malloc (sizeof (struct epoll_event) * epoll_eventmax);
00220     }
00221 
00222   /* now synthesize events for all fds where epoll fails, while select works... */
00223   for (i = epoll_epermcnt; i--; )
00224     {
00225       int fd = epoll_eperms [i];
00226       unsigned char events = anfds [fd].events & (EV_READ | EV_WRITE);
00227 
00228       if (anfds [fd].emask & EV_EMASK_EPERM && events)
00229         fd_event (EV_A_ fd, events);
00230       else
00231         {
00232           epoll_eperms [i] = epoll_eperms [--epoll_epermcnt];
00233           anfds [fd].emask = 0;
00234         }
00235     }
00236 }
