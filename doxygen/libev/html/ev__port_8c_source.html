00092 static void
00093 port_poll (EV_P_ ev_tstamp timeout)
00094 {
00095   int res, i;
00096   struct timespec ts;
00097   uint_t nget = 1;
00098 
00099   /* we initialise this to something we will skip in the loop, as */
00100   /* port_getn can return with nget unchanged, but no indication */
00101   /* whether it was the original value or has been updated :/ */
00102   port_events [0].portev_source = 0;
00103 
00104   EV_RELEASE_CB;
00105   EV_TS_SET (ts, timeout);
00106   res = port_getn (backend_fd, port_events, port_eventmax, &nget, &ts);
00107   EV_ACQUIRE_CB;
00108 
00109   /* port_getn may or may not set nget on error */
00110   /* so we rely on port_events [0].portev_source not being updated */
00111   if (res == -1 && errno != ETIME && errno != EINTR)
00112     ev_syserr ("(libev) port_getn (see http://bugs.opensolaris.org/view_bug.do?bug_id=6268715, try LIBEV_FLAGS=3 env variable)");
00113 
00114   for (i = 0; i < nget; ++i)
00115     {
00116       if (port_events [i].portev_source == PORT_SOURCE_FD)
00117         {
00118           int fd = port_events [i].portev_object;
00119 
00120           fd_event (
00121             EV_A_
00122             fd,
00123             (port_events [i].portev_events & (POLLOUT | POLLERR | POLLHUP) ? EV_WRITE : 0)
00124             | (port_events [i].portev_events & (POLLIN | POLLERR | POLLHUP) ? EV_READ : 0)
00125           );
00126 
00127           fd_change (EV_A_ fd, EV__IOFDSET);
00128         }
00129     }
00130 
00131   if (expect_false (nget == port_eventmax))
00132     {
00133       ev_free (port_events);
00134       port_eventmax = array_nextsize (sizeof (port_event_t), port_eventmax, port_eventmax + 1);
00135       port_events = (port_event_t *)ev_malloc (sizeof (port_event_t) * port_eventmax);
00136     }
00137 }

00139 int inline_size
00140 port_init (EV_P_ int flags)
00141 {
00142   /* Initialize the kernel queue */
00143   if ((backend_fd = port_create ()) < 0)
00144     return 0;
00145 
00146   assert (("libev: PORT_SOURCE_FD must not be zero", PORT_SOURCE_FD));
00147 
00148   fcntl (backend_fd, F_SETFD, FD_CLOEXEC); /* not sure if necessary, hopefully doesn't hurt */
00149 
00150   /* if my reading of the opensolaris kernel sources are correct, then
00151    * opensolaris does something very stupid: it checks if the time has already
00152    * elapsed and doesn't round up if that is the case,m otherwise it DOES round
00153    * up. Since we can't know what the case is, we need to guess by using a
00154    * "large enough" timeout. Normally, 1e-9 would be correct.
00155    */
00156   backend_mintime = 1e-3; /* needed to compensate for port_getn returning early */
00157   backend_modify  = port_modify;
00158   backend_poll    = port_poll;
00159 
00160   port_eventmax = 64; /* initial number of events receivable per poll */
00161   port_events = (port_event_t *)ev_malloc (sizeof (port_event_t) * port_eventmax);
00162 
00163   return EVBACKEND_PORT;
00164 }
