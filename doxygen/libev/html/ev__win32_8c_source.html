00063 ev_pipe (int filedes [2])
00064 {
00065   struct sockaddr_in addr = { 0 };
00066   int addr_size = sizeof (addr);
00067   struct sockaddr_in adr2;
00068   int adr2_size = sizeof (adr2);
00069   SOCKET listener;
00070   SOCKET sock [2] = { -1, -1 };
00071 
00072   if ((listener = ev_tcp_socket ()) == INVALID_SOCKET)
00073     return -1;
00074 
00075   addr.sin_family = AF_INET;
00076   addr.sin_addr.s_addr = htonl (INADDR_LOOPBACK);
00077   addr.sin_port = 0;
00078 
00079   if (bind (listener, (struct sockaddr *)&addr, addr_size))
00080     goto fail;
00081 
00082   if (getsockname (listener, (struct sockaddr *)&addr, &addr_size))
00083     goto fail;
00084 
00085   if (listen (listener, 1))
00086     goto fail;
00087 
00088   if ((sock [0] = ev_tcp_socket ()) == INVALID_SOCKET)
00089     goto fail;
00090 
00091   if (connect (sock [0], (struct sockaddr *)&addr, addr_size))
00092     goto fail;
00093 
00094   if ((sock [1] = accept (listener, 0, 0)) < 0)
00095     goto fail;
00096 
00097   /* windows vista returns fantasy port numbers for sockets:
00098    * example for two interconnected tcp sockets:
00099    *
00100    * (Socket::unpack_sockaddr_in getsockname $sock0)[0] == 53364
00101    * (Socket::unpack_sockaddr_in getpeername $sock0)[0] == 53363
00102    * (Socket::unpack_sockaddr_in getsockname $sock1)[0] == 53363
00103    * (Socket::unpack_sockaddr_in getpeername $sock1)[0] == 53365
00104    *
00105    * wow! tridirectional sockets!
00106    *
00107    * this way of checking ports seems to work:
00108    */
00109   if (getpeername (sock [0], (struct sockaddr *)&addr, &addr_size))
00110     goto fail;
00111 
00112   if (getsockname (sock [1], (struct sockaddr *)&adr2, &adr2_size))
00113     goto fail;
00114 
00115   errno = WSAEINVAL;
00116   if (addr_size != adr2_size
00117       || addr.sin_addr.s_addr != adr2.sin_addr.s_addr /* just to be sure, I mean, it's windows */
00118       || addr.sin_port        != adr2.sin_port)
00119     goto fail;
00120 
00121   closesocket (listener);
00122 
00123 #if EV_SELECT_IS_WINSOCKET
00124   filedes [0] = EV_WIN32_HANDLE_TO_FD (sock [0]);
00125   filedes [1] = EV_WIN32_HANDLE_TO_FD (sock [1]);
00126 #else
00127   /* when select isn't winsocket, we also expect socket, connect, accept etc.
00128    * to work on fds */
00129   filedes [0] = sock [0];
00130   filedes [1] = sock [1];
00131 #endif
00132 
00133   return 0;
00134 
00135 fail:
00136   closesocket (listener);
00137 
00138   if (sock [0] != INVALID_SOCKET) closesocket (sock [0]);
00139   if (sock [1] != INVALID_SOCKET) closesocket (sock [1]);
00140 
00141   return -1;
00142 }
