00186 void inline_size
00187 kqueue_fork (EV_P)
00188 {
00189   /* some BSD kernels don't just destroy the kqueue itself,
00190    * but also close the fd, which isn't documented, and
00191    * impossible to support properly.
00192    * we remember the pid of the kqueue call and only close
00193    * the fd if the pid is still the same.
00194    * this leaks fds on sane kernels, but BSD interfaces are
00195    * notoriously buggy and rarely get fixed.
00196    */
00197   pid_t newpid = getpid ();
00198 
00199   if (newpid == kqueue_fd_pid)
00200     close (backend_fd);
00201 
00202   kqueue_fd_pid = newpid;
00203   while ((backend_fd = kqueue ()) < 0)
00204     ev_syserr ("(libev) kqueue");
00205 
00206   fcntl (backend_fd, F_SETFD, FD_CLOEXEC);
00207 
00208   /* re-register interest in fds */
00209   fd_rearm_all (EV_A);
00210 }
