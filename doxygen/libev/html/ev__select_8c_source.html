00072 {
00073   if (oev == nev)
00074     return;
00075 
00076   {
00077 #if EV_SELECT_USE_FD_SET
00078 
00079     #if EV_SELECT_IS_WINSOCKET
00080     SOCKET handle = anfds [fd].handle;
00081     #else
00082     int handle = fd;
00083     #endif
00084 
00085     assert (("libev: fd >= FD_SETSIZE passed to fd_set-based select backend", fd < FD_SETSIZE));
00086 
00087     /* FD_SET is broken on windows (it adds the fd to a set twice or more,
00088      * which eventually leads to overflows). Need to call it only on changes.
00089      */
00090     #if EV_SELECT_IS_WINSOCKET
00091     if ((oev ^ nev) & EV_READ)
00092     #endif
00093       if (nev & EV_READ)
00094         FD_SET (handle, (fd_set *)vec_ri);
00095       else
00096         FD_CLR (handle, (fd_set *)vec_ri);
00097 
00098     #if EV_SELECT_IS_WINSOCKET
00099     if ((oev ^ nev) & EV_WRITE)
00100     #endif
00101       if (nev & EV_WRITE)
00102         FD_SET (handle, (fd_set *)vec_wi);
00103       else
00104         FD_CLR (handle, (fd_set *)vec_wi);
00105 
00106 #else
00107 
00108     int     word = fd / NFDBITS;
00109     fd_mask mask = 1UL << (fd % NFDBITS);
00110 
00111     if (expect_false (vec_max <= word))
00112       {
00113         int new_max = word + 1;
00114 
00115         vec_ri = ev_realloc (vec_ri, new_max * NFDBYTES);
00116         vec_ro = ev_realloc (vec_ro, new_max * NFDBYTES); /* could free/malloc */
00117         vec_wi = ev_realloc (vec_wi, new_max * NFDBYTES);
00118         vec_wo = ev_realloc (vec_wo, new_max * NFDBYTES); /* could free/malloc */
00119         #ifdef _WIN32
00120         vec_eo = ev_realloc (vec_eo, new_max * NFDBYTES); /* could free/malloc */
00121         #endif
00122 
00123         for (; vec_max < new_max; ++vec_max)
00124           ((fd_mask *)vec_ri) [vec_max] =
00125           ((fd_mask *)vec_wi) [vec_max] = 0;
00126       }
00127 
00128     ((fd_mask *)vec_ri) [word] |= mask;
00129     if (!(nev & EV_READ))
00130       ((fd_mask *)vec_ri) [word] &= ~mask;
00131 
00132     ((fd_mask *)vec_wi) [word] |= mask;
00133     if (!(nev & EV_WRITE))
